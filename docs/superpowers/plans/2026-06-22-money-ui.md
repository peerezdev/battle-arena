# Money UI — reserved/available balance + cancel lobby (#4c) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Show available-vs-reserved USDC in the header and let a lobby's creator cancel it (from the Hub rows and the battle flow), surfacing the money state the backend already tracks.

**Architecture:** A new authed `GET /users/me/balance` exposes `reserved_total`; `list_open` gains `creator_wallet`. The frontend adds a `useReservedBalance` poll hook, computes `available = max(0, on-chain − reserved)` in the header chip, and renders a creator-only Cancel control (wired to the existing `cancelBattle`) in the Hub rows and the `BattleFlow` waiting room.

**Tech Stack:** Backend FastAPI + SQLAlchemy (pytest, in-memory SQLite). Frontend React 19 + Vite + TypeScript, Privy auth, vitest + @testing-library/react.

## Global Constraints

- **Spec:** `docs/superpowers/specs/2026-06-22-money-ui-design.md`.
- **Backend tests** run from `backend/` with `.venv/bin/pytest`. **Frontend tests** run from repo root with `npm test`; typecheck `npx tsc --noEmit`.
- **No new dependencies.** Reuse `reserved_total`, `current_user`, `cancelBattle`, `useUsdcBalance`, `useIdentityToken`, `useEmbeddedSolanaAddress`, theme tokens.
- **Units:** `reserved` from the backend is **USDC base units** (×1e6). `useReservedBalance` converts to **dollars** (`/1e6`) to match `useUsdcBalance`. `available = usdc != null && reserved != null ? Math.max(0, usdc − reserved) : usdc`.
- **Reserved semantics:** active pack-lobby reservations only (creator + each joiner's `price`); royale money is escrowed (already reflected on-chain), not reserved.
- **Cancel:** creator-only, `lobby`-only; the backend `cancel_battle` enforces it (409 otherwise). `cancelBattle(token, id)` already exists.
- **Auth:** `current_user` raises 401 without a valid `Authorization: Bearer <identity token>`.
- **Out of scope (do NOT build):** PF verify panel (#4d), itemized per-lobby reserved breakdown, royale escrow "at stake" line, changes to reservation/refund/cancel logic, multi-pack (#4e).

```ts
import { useEffect, useRef, useState } from 'react'
import { useIdentityToken } from '@privy-io/react-auth'
import { fetchReservedBalance } from '../onchain/packBattleClient'

/** On-chain USDC minus reserved (both dollars); clamps at 0. Falls back to usdc when reserved unknown. */
export function availableUsd(usdc: number | null, reserved: number | null): number | null {
  if (usdc == null) return null
  if (reserved == null) return usdc
  return Math.max(0, usdc - reserved)
}

export function useReservedBalance(): { reserved: number | null } {
  const { identityToken } = useIdentityToken()
  const [reserved, setReserved] = useState<number | null>(null)
  const unmounted = useRef(false)

  useEffect(() => {
    unmounted.current = false
    return () => { unmounted.current = true }
  }, [])

  useEffect(() => {
    if (!identityToken) { setReserved(null); return }
    let timer: ReturnType<typeof setInterval> | null = null

    const poll = async () => {
      try {
        const { reserved: base } = await fetchReservedBalance(identityToken)
        if (!unmounted.current) setReserved(base / 1e6)   // base units → dollars
      } catch {
        if (!unmounted.current) setReserved(null)          // never block the balance display
      }
    }

    poll()
    timer = setInterval(poll, 15_000)
    return () => { if (timer) clearInterval(timer) }
  }, [identityToken])

  return { reserved }
}
```

```ts
export function openBattleToLive(b: OpenBattle, meWallet: string | null = null): LiveBattle {
  const shown = Math.min(b.players, MAX_AVATARS)

  return {

    mode: b.mode,
    live: false,
    title: b.machine_code,
    sub: `${b.players}/${b.max_players} joined`,
    players,
    extra,
    cards: b.mode === 'royale' ? ['🎴'] : ['🔥', '💧'],
    costLabel: b.mode === 'royale' ? 'ENTRY' : 'BUY-IN',
    costValue: b.buyin / BASE_UNITS,
    action: b.players < b.max_players ? 'join' : 'watch',
    canCancel: !!meWallet && b.creator_wallet === meWallet,
  }
}
```

```tsx
    })
  }
```
